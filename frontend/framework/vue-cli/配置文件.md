// vue.config.js 
// vue.config.js 运行在 Node.js 环境下 需要 commonjs 导入
const { version, appName } = require('./src/config.ts')
const { resolve } = require('path')

module.exports = {
	// 基础路由
	publicPath: '/',
	outputDir: 'dist',
	// 打包输出的静态文件
	assetsDir: 'static',
	// 禁用生产环境的 source maps，减少打包体积
	productionSourceMap: false,
	// 关闭 eslint 检查以提高构建速度
	// eslint: {
	// 	enable: false
	// },
	configureWebpack: {
		// 指定入口文件
		entry: './src/main.ts',
		resolve: {
			alias: {
				'@/': resolve(__dirname, 'src/')
			},
			extensions: ['.js', '.ts', '.vue', '.json'] // 扩展后缀名
		},
		performance: {
			hints: false // 关闭性能提示，构建时实际文件超出大小限制也不提示
			// maxEntrypointSize: 512000, //  将入口点大小限制增加到512KB,默认是244KB
			// maxAssetSize: 512000 //  将静态大小限制增加到512KB,默认是244KB
		},
		// 生产环境的优化配置
		optimization: {
			// 启用摇树优化
			usedExports: true,
			// 拆分公共模块
			splitChunks: {
				chunks: 'all',
				minSize: 30000, // 30KB，小于30KB的不分割
				maxSize: 244000, // 244KB，大于244KB的强制分割
				minChunks: 1, 
				automaticNameDelimiter: '.',
				cacheGroups: {
					// 分割 Ant Design Vue
					antd: {
						name: 'chunk-antd',
						test: /[\\/]node_modules[\\/](ant-design-vue|@ant-design)[\\/]/,
						priority: 10,
						enforce: true
					},
					// 分割编辑器相关依赖
					editor: {
						name: 'chunk-editor',
						test: /[\\/]node_modules[\\/](codemirror|vue-codemirror-next|mavon-editor)[\\/]/,
						priority: 9,
						enforce: true
					},
					// 其他第三方库
					vendors: {
						name: 'vendors',
						test: /[\\/]node_modules[\\/]/,
						priority: -10
					},
					// 公共代码
					common: {
						name: 'common',
						maxSize: 24400,
						minChunks: 3,
						priority: -20,
						reuseExistingChunk: true
					}
				}
			}
		}
	},
	// 链式操作 webpack 配置
	chainWebpack: (chainConfig) => {
		// 为 index.html 添加版本号
		chainConfig.plugin('html').tap((args) => {
			args[0].title = appName
			// 版本号
			args[0].version = version
			// 添加构建时间
			args[0].buildTime = new Date().toLocaleString('zh-CN')
			return args
		})

		// 配置JS压缩
		if (process.env.NODE_ENV === 'production') {
			// 压缩JS代码
			chainConfig.optimization.minimizer('terser').tap((args) => {
				args[0].terserOptions.compress.drop_console = true
				args[0].terserOptions.compress.drop_debugger = true
				return args
			})
		}
	},

	devServer: {
		host: '0.0.0.0',
		port: 8072,
		// open: true,
		hot: true, // 开启热更新，开启之后若页面还是没生效尝试重启项目或者重开一个标签页
		// compress: true,
		client: {
			overlay: false // 关闭浏览器中覆盖在页面上的错误遮罩，如提示 resize 回掉函数多次触发
		},
		proxy: {
			// 可以在这里添加代理配置
			// '/api': {
			//   target: 'http://api.example.com',
			//   changeOrigin: true,
			//   pathRewrite: { '^/api': '' }
			// }
		}
	},
	css: {
		loaderOptions: {
			less: {
				lessOptions: {
					// 启用 LESS 中的 JavaScript 表达式支持。
					javascriptEnabled: true
				},
                // 添加公用样式文件
				additionalData: `@import url("@/assets/styles/common.less");`
			}
		},
		// 提取CSS到单独的文件
		extract: {
			filename: 'static/css/[name].[contenthash:8].css',
			chunkFilename: 'static/css/[name].[contenthash:8].css'
		}
	}
}
