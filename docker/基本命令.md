```
# 查看系统的docker状态
systemctl status docker

# 打印现在运行的容器
docker container ls 

docker ps 

# 打印全部容器
docker ps -a

# 打景象
docker build

# 创建容器
docker container create

# 启动容器
docker container start xxx(容器ID)

# 停止容器
docker container stop xxx(容器ID)

# 重启容器
docker container restart xxx(容器ID)

# 创建并启动容器
docker run

# 打印镜像列表
docker images

# 更多指令
docker --help
```
